# 测试支持与源码索引

## 1. 测试支持

### 1.1 测试现状分析

WMS 模块的测试面临以下挑战：

| 挑战 | 说明 | 当前状态 |
|------|------|---------|
| 库存并发 | 入库/出库并发操作 | 需要并发测试 |
| 数量计算 | 库存增减、锁定解锁 | 需要精确断言 |
| 批次管理 | 批次号、生产日期、过期日期 | 需要数据准备 |
| 库位结构 | 仓库→库区→库位三级结构 | 需要层次化测试 |

#### 库存并发测试

```java
/**
 * 库存并发测试
 * 验证并发场景下的库存准确性
 */
public class WmsInventoryConcurrencyTest {

    @Test
    void testConcurrentInbound() throws InterruptedException {
        // 1. 准备数据
        Long warehouseId = createTestWarehouse();
        Long itemId = createTestItem();
        Long skuId = createTestSku(itemId);
        int threadCount = 10;
        BigDecimal quantityPerThread = BigDecimal.valueOf(100);
        
        // 2. 并发入库
        ExecutorService executor = Executors.newFixedThreadPool(threadCount);
        CountDownLatch latch = new CountDownLatch(threadCount);
        
        for (int i = 0; i < threadCount; i++) {
            executor.submit(() -> {
                try {
                    executeInbound(warehouseId, itemId, skuId, quantityPerThread);
                } finally {
                    latch.countDown();
                }
            });
        }
        
        latch.await();
        executor.shutdown();
        
        // 3. 验证总数量
        BigDecimal expectedTotal = quantityPerThread.multiply(BigDecimal.valueOf(threadCount));
        assertInventoryQuantity(warehouseId, itemId, skuId, expectedTotal);
    }

    @Test
    void testConcurrentOutbound() throws InterruptedException {
        // 1. 准备数据（先入库 1000）
        Long warehouseId = createTestWarehouse();
        Long itemId = createTestItem();
        Long skuId = createTestSku(itemId);
        executeInbound(warehouseId, itemId, skuId, BigDecimal.valueOf(1000));
        
        // 2. 并发出库（10 个线程，每个出库 100）
        int threadCount = 10;
        BigDecimal quantityPerThread = BigDecimal.valueOf(100);
        
        ExecutorService executor = Executors.newFixedThreadPool(threadCount);
        CountDownLatch latch = new CountDownLatch(threadCount);
        AtomicInteger successCount = new AtomicInteger(0);
        
        for (int i = 0; i < threadCount; i++) {
            executor.submit(() -> {
                try {
                    executeOutbound(warehouseId, itemId, skuId, quantityPerThread);
                    successCount.incrementAndGet();
                } catch (Exception e) {
                    // 库存不足时会抛出异常
                } finally {
                    latch.countDown();
                }
            });
        }
        
        latch.await();
        executor.shutdown();
        
        // 3. 验证库存（应该为 0）
        assertInventoryQuantity(warehouseId, itemId, skuId, BigDecimal.ZERO);
    }
}
```

```java
void testInventoryCheck() {
```

```java
@Resource
private WmsInboundOrderService inboundOrderService;

public Long createInboundOrder() {
    WmsInboundOrderSaveReqVO reqVO = new WmsInboundOrderSaveReqVO();
    reqVO.setCode("IN-2024-001");
```
