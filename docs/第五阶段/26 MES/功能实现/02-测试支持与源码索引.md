# 测试支持与源码索引

## 1. 测试支持

### 1.1 测试现状分析

MES 模块的测试面临以下挑战：

| 挑战 | 说明 | 当前状态 |
|------|------|---------|
| 业务流程复杂 | 工单→排产→报工→质检 多环节 | 需要集成测试 |
| 状态流转多 | 工单 5 种状态，流转规则严格 | 需要状态机测试 |
| 数量计算 | 已生产/已排产数量实时更新 | 需要并发测试 |
| 关联数据多 | 产品、客户、供应商、工序等 | 需要数据准备 |

```java
/**
 * MES 工单测试助手
 * 简化工单测试的数据准备和断言

public class MesWorkOrderTestHelper {

    @Resource
    private MesProWorkOrderService workOrderService;
    @Resource
    private MesProWorkOrderBomService workOrderBomService;
    @Resource
    private MesProReportService reportService;

    /**
     * 创建测试工单
     */
    public MesProWorkOrderDO createTestWorkOrder(Long productId, BigDecimal quantity) {
        MesProWorkOrderSaveReqVO reqVO = new MesProWorkOrderSaveReqVO();
        reqVO.setCode("WO-" + System.currentTimeMillis());
        reqVO.setName("测试工单");
        reqVO.setType(MesProWorkOrderTypeEnum.NORMAL.getType());
        reqVO.setProductId(productId);
        reqVO.setQuantity(quantity);
        reqVO.setStatus(MesProWorkOrderStatusEnum.DRAFT.getStatus());
        
        Long workOrderId = workOrderService.createWorkOrder(reqVO);
        return workOrderService.getWorkOrder(workOrderId);
    }

    /**
     * 创建测试 BOM
     */
    public List<MesProWorkOrderBomDO> createTestBom(Long workOrderId, Long itemId, BigDecimal quantity) {
        MesProWorkOrderBomSaveReqVO reqVO = new MesProWorkOrderBomSaveReqVO();
        reqVO.setWorkOrderId(workOrderId);
        reqVO.setItemId(itemId);
        reqVO.setQuantity(quantity);
        
        workOrderBomService.createWorkOrderBom(reqVO);
        return workOrderBomService.getWorkOrderBomListByWorkOrderId(workOrderId);
    }

    /**
     * 执行报工
     */
    public MesProReportDO executeReport(Long workOrderId, Long processId, BigDecimal quantity) {
        MesProReportSaveReqVO reqVO = new MesProReportSaveReqVO();
        reqVO.setCode("RPT-" + System.currentTimeMillis());
        reqVO.setWorkOrderId(workOrderId);
        reqVO.setProcessId(processId);
        reqVO.setQuantity(quantity);
        reqVO.setQuantityQualified(quantity);
        reqVO.setQuantityUnqualified(BigDecimal.ZERO);
        reqVO.setReportUserId(1L);
        reqVO.setReportTime(LocalDateTime.now());
        
        Long reportId = reportService.createReport(reqVO);
        return reportService.getReport(reportId);
    }

    /**
     * 断言工单状态
     */
    public void assertWorkOrderStatus(Long workOrderId, MesProWorkOrderStatusEnum expectedStatus) {
        MesProWorkOrderDO workOrder = workOrderService.getWorkOrder(workOrderId);
        assertEquals(expectedStatus.getStatus(), workOrder.getStatus());
    }

    /**
     * 断言工单已生产数量
     */
    public void assertQuantityProduced(Long workOrderId, BigDecimal expectedQuantity) {
        MesProWorkOrderDO workOrder = workOrderService.getWorkOrder(workOrderId);
        assertEquals(0, expectedQuantity.compareTo(workOrder.getQuantityProduced()));
    }
}
```

#### 工单状态机测试

```java
/**
 * 工单状态机测试
 * 验证状态流转的正确性
 */
public class MesWorkOrderStateMachineTest {

    @Test
    void testDraftToConfirmed() {
        // 1. 创建草稿工单
        MesProWorkOrderDO workOrder = createTestWorkOrder(1L, BigDecimal.valueOf(100));
        assertEquals(MesProWorkOrderStatusEnum.DRAFT.getStatus(), workOrder.getStatus());
        
        // 2. 确认工单
        workOrderService.confirmWorkOrder(workOrder.getId());
        
        // 3. 验证状态
        assertWorkOrderStatus(workOrder.getId(), MesProWorkOrderStatusEnum.CONFIRMED);
    }

    @Test
    void testConfirmedToInProgress() {
        // 1. 创建并确认工单
        MesProWorkOrderDO workOrder = createTestWorkOrder(1L, BigDecimal.valueOf(100));
        workOrderService.confirmWorkOrder(workOrder.getId());
        
        // 2. 模拟开始生产（通过排产或手动更新）
        workOrderService.updateWorkOrderStatus(workOrder.getId(), MesProWorkOrderStatusEnum.IN_PROGRESS);
        
        // 3. 验证状态
        assertWorkOrderStatus(workOrder.getId(), MesProWorkOrderStatusEnum.IN_PROGRESS);
    }

    @Test
    void testInProgressToFinished() {
        // 1. 创建进行中的工单
        MesProWorkOrderDO workOrder = createTestWorkOrder(1L, BigDecimal.valueOf(100));
        workOrderService.confirmWorkOrder(workOrder.getId());
        workOrderService.updateWorkOrderStatus(workOrder.getId(), MesProWorkOrderStatusEnum.IN_PROGRESS);
        
        // 2. 报工完成
        executeReport(workOrder.getId(), 1L, BigDecimal.valueOf(100));
        
        // 3. 验证状态（自动完成）
        assertWorkOrderStatus(workOrder.getId(), MesProWorkOrderStatusEnum.FINISHED);
    }

    @Test
    void testCancelFromDraft() {
        // 1. 创建草稿工单
        MesProWorkOrderDO workOrder = createTestWorkOrder(1L, BigDecimal.valueOf(100));
        
        // 2. 取消工单
        workOrderService.cancelWorkOrder(workOrder.getId());
        
        // 3. 验证状态
        assertWorkOrderStatus(workOrder.getId(), MesProWorkOrderStatusEnum.CANCELLED);
    }

    @Test
    void testInvalidTransition() {
        // 1. 创建草稿工单
        MesProWorkOrderDO workOrder = createTestWorkOrder(1L, BigDecimal.valueOf(100));
        
        // 2. 尝试直接完成（应该失败）
        assertThrows(ServiceException.class, () -> {
            workOrderService.finishWorkOrder(workOrder.getId());
        });
    }
}
```

#### 报工数量测试

```java
/**
 * 报工数量测试
 * 验证数量计算的正确性
 */
public class MesReportQuantityTest {

    @Test
    void testQuantityAccumulation() {
        // 1. 创建工单（数量 100）
        MesProWorkOrderDO workOrder = createTestWorkOrder(1L, BigDecimal.valueOf(100));
        workOrderService.confirmWorkOrder(workOrder.getId());
        workOrderService.updateWorkOrderStatus(workOrder.getId(), MesProWorkOrderStatusEnum.IN_PROGRESS);
        
        // 2. 第一次报工（30）
        executeReport(workOrder.getId(), 1L, BigDecimal.valueOf(30));
        assertQuantityProduced(workOrder.getId(), BigDecimal.valueOf(30));
        
        // 3. 第二次报工（50）
        executeReport(workOrder.getId(), 1L, BigDecimal.valueOf(50));
        assertQuantityProduced(workOrder.getId(), BigDecimal.valueOf(80));
        
        // 4. 第三次报工（20）- 完成
        executeReport(workOrder.getId(), 1L, BigDecimal.valueOf(20));
        assertQuantityProduced(workOrder.getId(), BigDecimal.valueOf(100));
        
        // 5. 验证工单状态（自动完成）
        assertWorkOrderStatus(workOrder.getId(), MesProWorkOrderStatusEnum.FINISHED);
    }

    @Test
    void testOverReport() {
        // 1. 创建工单（数量 100）
        MesProWorkOrderDO workOrder = createTestWorkOrder(1L, BigDecimal.valueOf(100));
        workOrderService.confirmWorkOrder(workOrder.getId());
        workOrderService.updateWorkOrderStatus(workOrder.getId(), MesProWorkOrderStatusEnum.IN_PROGRESS);
        
        // 2. 尝试报工 120（超过工单数量）
        assertThrows(ServiceException.class, () -> {
            executeReport(workOrder.getId(), 1L, BigDecimal.valueOf(120));
        });
    }
}
```

```java
@Test
void testGenerateWorkOrderBom() {
    // 1. 创建产品和产品 BOM
    Long productId = createTestProduct();
    createTestProductBom(productId, Arrays.asList(
        new BomLine(itemId1, BigDecimal.valueOf(2)),

    ));
    
    // 2. 创建工单（数量 100）
    MesProWorkOrderDO workOrder = createTestWorkOrder(productId, BigDecimal.valueOf(100));
    
    // 3. 验证 BOM 自动生成

    // 4. 验证数量计算（物料用量 * 工单数量）
    MesProWorkOrderBomDO bom1 = bomList.stream()
        .filter(b -> b.getItemId().equals(itemId1))
        .findFirst().orElseThrow();
    assertEquals(0, BigDecimal.valueOf(200).compareTo(bom1.getQuantity())); // 2 * 100
    
    MesProWorkOrderBomDO bom2 = bomList.stream()
        .filter(b -> b.getItemId().equals(itemId2))
        .findFirst().orElseThrow();
    assertEquals(0, BigDecimal.valueOf(300).compareTo(bom2.getQuantity())); // 3 * 100
}
```
