### 1.2 创建流程实例

```java
// 来源：yudao-module-bpm-server/.../service/task/BpmProcessInstanceServiceImpl.java
@Override
@Transactional(rollbackFor = Exception.class)
public String createProcessInstance(Long userId, BpmProcessInstanceCreateReqVO createReqVO) {
    // 1. 获取流程定义
    ProcessDefinition processDefinition = processDefinitionService.getProcessDefinition(
            createReqVO.getProcessDefinitionId());
    if (processDefinition == null) {
        throw exception(PROCESS_DEFINITION_NOT_EXISTS);
    }

    // 2. 校验流程定义是否激活
    if (processDefinition.isSuspended()) {
        throw exception(PROCESS_DEFINITION_IS_SUSPENDED);
    }

    // 3. 构建流程变量
    Map<String, Object> variables = new HashMap<>();
    variables.put(BpmnVariableConstants.PROCESS_INSTANCE_VARIABLE_START_USER_ID, userId);
    if (CollUtil.isNotEmpty(createReqVO.getVariables())) {
        variables.putAll(createReqVO.getVariables());
    }

    // 4. 启动流程实例
    ProcessInstanceBuilder builder = runtimeService.createProcessInstanceBuilder()
            .processDefinitionId(processDefinition.getId())
            .variables(variables)
            .tenantId(FlowableUtils.getTenantId());

    // 设置发起人
    String startUserId = String.valueOf(userId);
    builder.startUserId(startUserId);

    ProcessInstance processInstance = builder.start();

    // 5. 保存表单数据
    if (createReqVO.getFormType() != null && createReqVO.getFormType().equals(
            BpmModelFormTypeEnum.BPM_FORM.getType())) {
        // 内置动态表单：保存到流程变量
        runtimeService.setVariables(processInstance.getId(), createReqVO.getVariables());
    } else if (createReqVO.getFormType() != null && createReqVO.getFormType().equals(
            BpmModelFormTypeEnum.CUSTOM.getType())) {
        // 业务表单：由业务模块自己处理
    }

    // 6. 发布流程实例创建事件
    processInstanceEventPublisher.publishProcessInstanceCreatedEvent(processInstance);

    return processInstance.getId();
}
```

设计要点：
- **流程变量**：使用 `variables` 传递流程数据，用于条件分支判断和审批人选择。
- **发起人**：通过 `startUserId` 设置流程发起人，用于后续的"发起人自选审批人"功能。
- **表单数据**：支持两种存储方式——内置动态表单存储到流程变量，业务表单由业务模块自己处理。

### 1.3 获取审批详情

```java
@Override
public BpmApprovalDetailRespVO getApprovalDetail(Long loginUserId, BpmApprovalDetailReqVO reqVO) {
    // 1. 获取流程实例信息
    Long startUserId = loginUserId;
    HistoricProcessInstance historicProcessInstance = null;
    Integer processInstanceStatus = BpmProcessInstanceStatusEnum.NOT_START.getStatus();
    Map<String, Object> processVariables = new HashMap<>();

    if (reqVO.getProcessInstanceId() != null) {
        historicProcessInstance = getHistoricProcessInstance(reqVO.getProcessInstanceId());
        startUserId = Long.valueOf(historicProcessInstance.getStartUserId());
        processInstanceStatus = FlowableUtils.getProcessInstanceStatus(historicProcessInstance);
        if (CollUtil.isNotEmpty(historicProcessInstance.getProcessVariables())) {
            processVariables.putAll(historicProcessInstance.getProcessVariables());
        }
    }

    // 2. 获取流程定义信息
    ProcessDefinition processDefinition = processDefinitionService.getProcessDefinition(
            historicProcessInstance != null ? historicProcessInstance.getProcessDefinitionId()
                    : reqVO.getProcessDefinitionId());
    BpmProcessDefinitionInfoDO processDefinitionInfo = processDefinitionService
            .getProcessDefinitionInfo(processDefinition.getId());
    BpmnModel bpmnModel = processDefinitionService.getProcessDefinitionBpmnModel(processDefinition.getId());

    // 3. 构建审批节点列表
    List<ActivityNode> endActivityNodes = null; // 已结束的节点
    List<ActivityNode> runActivityNodes = null; // 进行中的节点
    if (reqVO.getProcessInstanceId() != null) {
        List<HistoricActivityInstance> activities = taskService.getActivityListByProcessInstanceId(
                reqVO.getProcessInstanceId());
        List<HistoricTaskInstance> tasks = taskService.getTaskListByProcessInstanceId(
                reqVO.getProcessInstanceId(), true);
        endActivityNodes = getEndActivityNodeList(startUserId, bpmnModel, processDefinitionInfo,
                historicProcessInstance, processInstanceStatus, activities, tasks);
        runActivityNodes = getRunApproveNodeList(startUserId, bpmnModel, processDefinition,
                processVariables, activities, tasks);
    }

    // 4. 预测未运行节点
    List<ActivityNode> simulateActivityNodes = null;
    if (!BpmProcessInstanceStatusEnum.isProcessEndStatus(processInstanceStatus)) {
        simulateActivityNodes = getSimulateApproveNodeList(startUserId, bpmnModel, processDefinition,
                processVariables, endActivityNodes, runActivityNodes);
    }

    // 5. 构建返回结果
    return buildApprovalDetail(reqVO, bpmnModel, processDefinition, processDefinitionInfo,
            historicProcessInstance, processInstanceStatus, endActivityNodes, runActivityNodes,
            simulateActivityNodes, todoTask);
}
```

### 2.2 审批通过

```java
// 来源：yudao-module-bpm-server/.../service/task/BpmTaskServiceImpl.java
@Override
@Transactional(rollbackFor = Exception.class)
@DataPermission(enable = false) // 关闭数据权限
public void approveTask(Long userId, @Valid BpmTaskApproveReqVO reqVO) {
    // 1. 校验任务
    Task task = validateTask(userId, reqVO.getId());
    ProcessInstance instance = processInstanceService.getProcessInstance(task.getProcessInstanceId());
    if (instance == null) {
        throw exception(PROCESS_INSTANCE_NOT_EXISTS);
    }

    // 2. 校验签名（如果需要）
    BpmnModel bpmnModel = modelService.getBpmnModelByDefinitionId(task.getProcessDefinitionId());
    Boolean signEnable = parseSignEnable(bpmnModel, task.getTaskDefinitionKey());
    if (signEnable && StrUtil.isEmpty(reqVO.getSignPicUrl())) {
        throw exception(TASK_SIGNATURE_NOT_EXISTS);
    }

    // 3. 校验审批意见（如果需要）
    Boolean reasonRequire = parseReasonRequire(bpmnModel, task.getTaskDefinitionKey());
    if (reasonRequire && StrUtil.isEmpty(reqVO.getReason())) {
        throw exception(TASK_REASON_REQUIRE);
    }

    // 4. 处理被委派的任务
    if (DelegationState.PENDING.equals(task.getDelegationState())) {
        approveDelegateTask(reqVO, task);
        return;
    }

    // 5. 处理后加签的任务
    if (BpmTaskSignTypeEnum.AFTER.getType().equals(task.getScopeType())) {
        approveAfterSignTask(task, reqVO);
        return;
    }

    // 6. 普通任务审批
    // 6.1 更新任务状态
    updateTaskStatusAndReason(task.getId(), BpmTaskStatusEnum.APPROVE.getStatus(), reqVO.getReason());
    if (signEnable) {
        taskService.setVariableLocal(task.getId(), BpmnVariableConstants.TASK_SIGN_PIC_URL,
                reqVO.getSignPicUrl());
    }

    // 6.2 添加评论
    taskService.addComment(task.getId(), task.getProcessInstanceId(),
            BpmCommentTypeEnum.APPROVE.getType(),
            BpmCommentTypeEnum.APPROVE.formatComment(reqVO.getReason()));

    // 6.3 合并流程变量
    Map<String, Object> processVariables = new HashMap<>();
    if (CollUtil.isNotEmpty(instance.getProcessVariables())) {
        processVariables.putAll(instance.getProcessVariables());
    }
    if (CollUtil.isNotEmpty(reqVO.getVariables())) {
        processVariables.putAll(reqVO.getVariables());
    }

    // 6.4 校验并设置下一个节点的审批人
    Map<String, Object> variables = validateAndSetNextAssignees(
            task.getTaskDefinitionKey(), processVariables, bpmnModel,
            reqVO.getNextAssignees(), instance);
    runtimeService.setVariables(task.getProcessInstanceId(), variables);

    // 6.5 完成任务
    taskService.complete(task.getId(), variables, true);

    // 6.6 处理加签任务
    handleParentTaskIfSign(task.getParentTaskId());
}
```

设计要点：
- **流程变量合并**：先获取历史流程变量，再合并前端传递的变量，以前端为准。
- **下一个节点审批人**：支持"发起人自选"和"审批人选择"两种模式。
- **签名和审批意见**：可配置是否必填。
- **委派任务处理**：如果任务是被委派的，审批后需要回到原审批人。

### 2.3 审批拒绝

```java
@Override
@Transactional(rollbackFor = Exception.class)
public void rejectTask(Long userId, @Valid BpmTaskRejectReqVO reqVO) {
    // 1. 校验任务
    Task task = validateTask(userId, reqVO.getId());

    // 2. 更新任务状态
    updateTaskStatusAndReason(task.getId(), BpmTaskStatusEnum.REJECT.getStatus(), reqVO.getReason());

    // 3. 添加评论
    taskService.addComment(task.getId(), task.getProcessInstanceId(),
            BpmCommentTypeEnum.REJECT.getType(),
            BpmCommentTypeEnum.REJECT.formatComment(reqVO.getReason()));

    // 4. 更新流程实例为不通过
    processInstanceService.updateProcessInstanceReject(
            processInstanceService.getProcessInstance(task.getProcessInstanceId()),
            reqVO.getReason());
}
```

## 三、任务委派

### 3.1 委派流程

委派是将任务交给其他人处理，处理完后回到原审批人手中。

```
原审批人 A → 委派给 B → B 处理 → 回到 A → A 审批
```

```java
@Override
@Transactional(rollbackFor = Exception.class)
public void delegateTask(Long userId, BpmTaskDelegateReqVO reqVO) {
    // 1. 校验任务
    Task task = validateTask(userId, reqVO.getId());

    // 2. 执行委派
    taskService.delegateTask(task.getId(), String.valueOf(reqVO.getDelegateUserId()));

    // 3. 更新任务状态
    updateTaskStatus(task.getId(), BpmTaskStatusEnum.DELEGATE.getStatus());
}
```

### 3.2 委派任务审批

```java
private void approveDelegateTask(BpmTaskApproveReqVO reqVO, Task task) {
    // 1. 更新任务状态
    updateTaskStatusAndReason(task.getId(), BpmTaskStatusEnum.APPROVE.getStatus(), reqVO.getReason());

    // 2. 添加评论
    taskService.addComment(task.getId(), task.getProcessInstanceId(),
            BpmCommentTypeEnum.APPROVE.getType(),
            BpmCommentTypeEnum.APPROVE.formatComment(reqVO.getReason()));

    // 3. 完成任务（resolve 操作会让任务回到原审批人）
    taskService.resolveTask(task.getId());
}
```

## 四、任务转办

### 4.1 转办流程

转办是将任务永久转给其他人处理，原审批人不再参与。

```
原审批人 A → 转办给 B → B 处理
```

```java
@Override
@Transactional(rollbackFor = Exception.class)
public void transferTask(Long userId, BpmTaskTransferReqVO reqVO) {
    // 1. 校验任务
    Task task = validateTask(userId, reqVO.getId());

    // 2. 执行转办
    taskService.setAssignee(task.getId(), String.valueOf(reqVO.getAssigneeUserId()));

    // 3. 更新任务状态
    updateTaskStatus(task.getId(), BpmTaskStatusEnum.TRANSFER.getStatus());

    // 4. 添加评论
    taskService.addComment(task.getId(), task.getProcessInstanceId(),
            BpmCommentTypeEnum.TRANSFER.getType(),
            BpmCommentTypeEnum.TRANSFER.formatComment(reqVO.getReason()));
}
```

## 五、任务退回

### 5.1 退回流程

退回到指定的前置节点，用户需要重新处理。

```java
@Override
@Transactional(rollbackFor = Exception.class)
public void returnTask(Long userId, BpmTaskReturnReqVO reqVO) {
    // 1. 校验任务
    Task task = validateTask(userId, reqVO.getId());

    // 2. 获取可退回的节点
    List<UserTask> returnTasks = getUserTaskListByReturn(task.getId());
    boolean canReturn = returnTasks.stream()
            .anyMatch(userTask -> userTask.getId().equals(reqVO.getTargetDefinitionKey()));
    if (!canReturn) {
        throw exception(TASK_RETURN_FAIL);
    }

    // 3. 执行退回
    // 3.1 删除当前任务
    taskService.complete(task.getId(), new HashMap<>(), true);

    // 3.2 在目标节点创建新任务
    runtimeService.createChangeActivityStateBuilder()
            .processInstanceId(task.getProcessInstanceId())
            .moveActivityIdTo(task.getTaskDefinitionKey(), reqVO.getTargetDefinitionKey())
            .changeState();

    // 4. 更新任务状态
    updateTaskStatus(task.getId(), BpmTaskStatusEnum.RETURN.getStatus());
}
```

## 六、加签与减签

### 6.1 加签

加签分为"向前加签"和"向后加签"：

- **向前加签**：在当前审批人之前插入新的审批人，新审批人先处理，然后回到当前审批人
- **向后加签**：在当前审批人之后插入新的审批人，当前审批人先处理，然后新审批人处理

```java
@Override
@Transactional(rollbackFor = Exception.class)
public void createSignTask(Long userId, BpmTaskSignCreateReqVO reqVO) {
    // 1. 校验任务
    Task task = validateTask(userId, reqVO.getId());

    // 2. 创建子任务
    for (Long assigneeUserId : reqVO.getUserIds()) {
        TaskEntityImpl newTask = (TaskEntityImpl) taskService.newTask();
        newTask.setParentTaskId(task.getId());
        newTask.setProcessInstanceId(task.getProcessInstanceId());
        newTask.setProcessDefinitionId(task.getProcessDefinitionId());
        newTask.setTaskDefinitionKey(task.getTaskDefinitionKey());
        newTask.setName(task.getName());
        newTask.setAssignee(String.valueOf(assigneeUserId));
        newTask.setScopeType(reqVO.getSignType().toString()); // 标记加签类型
        taskService.saveTask(newTask);
    }

    // 3. 如果是向前加签，暂停当前任务
    if (BpmTaskSignTypeEnum.BEFORE.getType().equals(reqVO.getSignType())) {
        taskService.suspendTask(task.getId());
    }
}
```

### 6.2 减签

```java

            BpmCommentTypeEnum.DELETE_SIGN.formatComment(reqVO.getReason()));
}
```

## 七、任务撤回

### 7.1 撤回流程

发起人可以在下一个审批人未处理时撤回流程。

```java
@Override
@Transactional(rollbackFor = Exception.class)
public void withdrawTask(Long userId, String taskId) {
    // 1. 获取历史任务
    HistoricTaskInstance historicTask = getHistoricTask(taskId);
    if (historicTask == null) {
        throw exception(TASK_NOT_EXISTS);
    }

    // 2. 校验是否是发起人
    HistoricProcessInstance processInstance = historyService
            .createHistoricProcessInstanceQuery()
            .processInstanceId(historicTask.getProcessInstanceId())
            .singleResult();
    if (!String.valueOf(userId).equals(processInstance.getStartUserId())) {
        throw exception(TASK_WITHDRAW_FAIL_NOT_START_USER);
    }

    // 3. 校验下一个任务是否已处理
    List<Task> currentTasks = getRunningTaskListByProcessInstanceId(
            historicTask.getProcessInstanceId(), true, null);
    for (Task currentTask : currentTasks) {
        if (currentTask.getTaskDefinitionKey().equals(historicTask.getTaskDefinitionKey())) {
            // 下一个任务还在，可以撤回
            taskService.complete(currentTask.getId(), new HashMap<>(), true);
            break;
        }
    }

    // 4. 更新历史任务状态
    updateHistoricTaskStatus(taskId, BpmTaskStatusEnum.WITHDRAW.getStatus());
}
```

```java
@Override

public Long createLeave(Long userId, BpmOALeaveCreateReqVO reqVO) {
    // 1. 创建请假单

                    .setBusinessKey(String.valueOf(leave.getId())));

    leave.setProcessInstanceId(processInstanceId);
    leave.setStatus(BpmProcessInstanceStatusEnum.RUNNING.getStatus());
    leaveMapper.updateById(leave);

    return leave.getId();
}
```

### 8.3 流程状态监听

```java
// 来源：yudao-module-bpm-server/.../service/oa/listener/BpmOALeaveStatusListener.java
@Component
public class BpmOALeaveStatusListener {

    @Resource
    private BpmOALeaveService leaveService;

    @EventListener
    public void onProcessInstanceCompleted(BpmProcessInstanceCompleteEvent event) {
        // 1. 检查是否是 OA 请假流程
        ProcessInstance processInstance = event.getProcessInstance();
        if (!"oa_leave".equals(processInstance.getProcessDefinitionKey())) {
            return;
        }

        // 2. 获取请假单 ID
        String businessKey = processInstance.getBusinessKey();
        if (StrUtil.isEmpty(businessKey)) {
            return;
        }

        // 3. 更新请假单状态
        Integer status;
        if (processInstance.isEnded()) {
            status = BpmProcessInstanceStatusEnum.APPROVE.getStatus();
        } else {
            status = BpmProcessInstanceStatusEnum.REJECT.getStatus();
        }
        leaveService.updateLeaveStatus(Long.valueOf(businessKey), status);
    }
}
```

设计要点：
- **业务关联**：通过 `businessKey` 将流程实例与业务数据关联。
- **事件驱动**：使用 Spring 的 `@EventListener` 监听流程完成事件，更新业务状态。
- **解耦**：业务模块不需要主动轮询流程状态，而是被动接收事件通知。
